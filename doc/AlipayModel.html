def invalid_sign?
  attrs = @attributes.dup
  attrs.delete("sign")
  attrs.delete("sign_type")
  text = attrs.delete_if{ |k, v| v.to_s.length == 0 }.sort.collect{ |s| s[0] + "=" + URI.decode(s[1]) }.join("&") + self.key
  self.sign == Digest::MD5.hexdigest(text) ? false : true
end
