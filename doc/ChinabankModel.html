def invalid_sign?
  text = @attributes["v_amount"]+@attributes["v_moneytype"]+@attributes["v_oid"]+@attributes["v_mid"]+@attributes["v_url"]+self.key
  self.sign.upcase == Digest::MD5.hexdigest(text).upcase ? false : true
  rescue => e
  true
end
